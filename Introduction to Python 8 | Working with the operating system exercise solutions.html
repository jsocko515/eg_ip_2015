Kmer counting¶

To figure out which kmers occur more than the threshold number of times, we have to add up all the kmers for all the sequences in all the files before we know. So we need to build up the counts first, then process them.

We'll start with a function to split a DNA sequence into kmers of a given length:

In [13]:
def split_dna(dna, kmer_size): 
    kmers = [] 
    for start in range(0,len(dna)-(kmer_size-1),1): 
        kmer = dna[start:start+kmer_size] 
        kmers.append(kmer) 
    return kmers

        kmers.append(kmer)

        print("reading sequences from " + file_name) 
        dna_file = open("dna_files/" + file_name) 
        for line in dna_file: 
            dna = line.rstrip("\n") 
            for kmer in split_dna(dna, kmer_size): 
                current_count = kmer_counts.get(kmer, 0) 
                new_count = current_count + 1 
                kmer_counts[kmer] = new_count

def split_dna(dna, kmer_size): 
    kmers = [] 
    for start in range(0,len(dna)-(kmer_size-1),1): 
        kmer = dna[start:start+kmer_size] 
        kmers.append(kmer) 
    return kmers 
 
kmer_counts = {}
